<!--
  onsubmit="return false;"    阻止默认事件（必须）
  stverify="true"             标记验证表单（必须）
  erroappend="true"           元素后附加错误信息
  dynamicForm="true"          动态表单
  ajaxpost="true"             ajax提交
  beforeSubFun="_beforeSubFun"定义验证前执行的方法
  SubFun="_SubFun"            定义验证成功后提交前执行的方法
  afterSubFun="_afterSubFun"  定义验证成功后提交后执行的方法
  errorFun="_errorFun"        定义ajax提交失败后执行的方法
  confirm="true"              是否在提交前显示确认框
  conftmp="form_confmsg_temp" 确认框模板
  confmsg="确认提交吗？"       确认框内容
-->
  ...
</form>

<!--JS部分-->
<script src="__YOURPATH__/js/ST.Verify.js"></script>
<script>
  $.extend(ST,{
    _beforeSubFun:function(){
      alert('验证前执行的方法!返回true才能验证');
      return true;
    },
    _SubFun:function(){
      alert('验证成功后提交前执行的方法!返回true才能提交');
      return true;
    },
    _afterSubFun:function(){
      alert('验证成功后提交后执行的方法!');
    },
    _errorFun:function(){
      alert('ajax提交失败后执行的方法!');
    }
  });
</script>
