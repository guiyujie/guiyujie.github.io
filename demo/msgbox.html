弹出层 [code] 
<script>
ST.msgbox(
  {
    title:'弹出层',                 //标题
    content:'这是内容！'            //内容（支持HTML）
  },
  [                                //按钮定义
    {
      text:'自定义按钮',            //按钮名称
      fun:function(a,e){           //click后回调方法
        //设置点击后不关闭
        e.cancle = true;
        alert('这是自定义回调事件！');
      }
    },{text:'关闭',fun:''}
  ],
  480,                              //宽度（单位px）
  360,                              //高度（单位px）
  true                              //是否使用遮罩
);
</script>
弹出frame [code] 
<script>
ST.$Fb = ST.msgbox(
  {
    title:'弹出frame',              //标题
    url:'http://www.baidu.com'     //url
  },
  [                                //按钮定义
    {
      text:'自定义按钮',            //按钮名称
      fun:function(a,e){           //click后回调方法
        //设置点击后不关闭
        e.cancle = true;
        alert('这是自定义回调事件！');
      }
    },{text:'关闭',fun:''}
  ],
  800,                              //宽度（单位px）
  600,                              //高度（单位px）
  false                             //是否使用遮罩
);
/*
从子页面关闭窗口，调用：parent.ST.$Fb.hide();
从子页面刷新窗口，调用：parent.ST.reload();
*/
</script>
模拟confirm [code] 
<script>
//ST框架基于ST.msgbox实现了ST.confirm方法
ST.confirm(
  '<p class="m20">确定执行此操作吗？</p>', //内容（支持HTML）
  '操作提示',                             //标题
  function(){                            //确定操作执行的方法
    alert("执行了确认操作！");
  },
  function(){                            //取消操作执行的方法
    alert("执行了取消操作！");
  },
  240,                                   //宽度（单位px）
  180                                    //高度（单位px）
);
</script>
