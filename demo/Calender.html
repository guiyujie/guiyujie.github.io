<!--JS部分-->

<script>
  var this_year = new Date().getFullYear();
  /*
    参数：
    -------------------------------
    目标元素ID
    位　  置　     1上2右3下4左，默认3
    开始年份       默认1960
    结束年份       默认今年
    -------------------------------
  */
  var cal = new ST.Calender("_calender", "", this_year - 1, this_year);
  cal.onselected = function (c) {
    alert('您选择了：' + c.date);
  }
  //若在弹窗中使用请取消下一行注释
  //cal.zIndex = 9999;
</script>

<!--HTML部分-->
<input class="input input-small" type="text" id="_calender" readonly="readonly"/>
<script src="__YOURPATH__/js/ST.TimePicker.js"></script>
<script src="__YOURPATH__/js/ST.Calender.js"></script>
<!--JS部分-->
<script>
  /*
    参数：
    -------------------------------
    目标元素ID
    位　  置　     1上2右3下4左，默认3
    开始年份       默认1960
    结束年份       默认今年
    -------------------------------
  */
  var this_year = new Date().getFullYear();

  var cal = new ST.Calender("_calender", "", this_year - 1, this_year);
  cal.onselected = function (c) {
    alert('您选择了：' + c.date);
  }
  //设置时间选择
  cal.timePicker = true;  //是否显示时间
  cal.showSeconds = true; //是否显示秒
  //若在弹窗中使用请取消下一行注释
  //cal.zIndex = 9999;
</script>

    endYear: ST.CalRange.endYear,     //最大年份

});
