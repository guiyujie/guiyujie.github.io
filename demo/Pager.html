前端分页 [code] 
  <!--HTML部分-->
  <div id="pager_content"></div>
  <div id="pager"></div>

  <!--JS部分-->
  <script>
      //虚拟数据
      var _data = function () {
        var a = [];
        for (var i = 1; i <= 100; i++) {
          a.push({id: i, title: '这是第'+i+'条数据'});
        }
        return a;
      }();
      //填充内容
      var fillContent = function(page){
        var data = _data.slice((page - 1) * 10, page * 10), contentArr = [];
        for (var i = 0, l = data.length, d; i < l, d = data[i]; i++){
          contentArr.push('<p class="m5">'+ d.title+'</p>');
        }
        $("#pager_content").html(contentArr.join(''));
      };
      /*
        默认配置
       {
          dataNum:  数据条数
          pageSize: 页大小
          dynamic:   动态分页
          pageTemp:  分页模板
          pageNumTemp: 分页页码模板
          pageHash: 是否使用页面HASH（默认不使用）
       }
      */
      //初始化插件
      var page = ST.getHash('page');//获取页面hash
      var pager = new ST.Pager("pager", {dataNum: _data.length, pageSize: 10, pageHash: true}, function (page) {
        fillContent(page);
      });
      fillContent(page);
  </script>
